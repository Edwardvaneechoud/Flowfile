import flowfile as ff

df = ff.read_csv("data.csv")
result = df.filter(
    ff.col("amount") > 1000
).group_by("region").agg(
    ff.col("amount").sum()
)

# Visualize your pipeline
ff.open_graph_in_editor(result.flow_graph)
